import http from "http";

export function startHealthServer(port) {
  const server = http.createServer((req, res) => {
    if (req.url === "/health") {
      res.writeHead(200, { "Content-Type": "text/plain" });
      return res.end("ok");
    }
    res.writeHead(200, { "Content-Type": "text/plain" });
    res.end("cdl-bot");
  });

  server.listen(port, "0.0.0.0", () => {
    console.log(`Health server: ${port}`);
  });
}
